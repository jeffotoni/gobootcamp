API REST - Chi

chi é um roteador leve, idiomático e combinável para construir serviços Go HTTP. É especialmente bom para ajudá-lo a escrever grandes serviços de API REST que podem ser mantidos à medida que seu projeto cresce e muda. chi é construído no novo pacote de contexto introduzido no Go 1.7 para lidar com sinalização, cancelamento e valores com escopo de solicitação em uma cadeia de manipulador.

chi.go

package main

import (
	"net/http"

	"github.com/go-chi/chi/v5"
	"github.com/go-chi/chi/v5/middleware"
)

func main() {
	r := chi.NewRouter()
	r.Use(middleware.Logger)

	r.Get("/v1/api", func(w http.ResponseWriter, r *http.Request) {
		w.Write([]byte("GETTING..."))
	})

    r.Post("/v1/api", func(w http.ResponseWriter, r *http.Request) {
		w.Write([]byte("POSTING..."))
	})

    r.Put("/v1/api", func(w http.ResponseWriter, r *http.Request) {
		w.Write([]byte("PUTTING..."))
	})

    r.Delete("/v1/api", func(w http.ResponseWriter, r *http.Request) {
		w.Write([]byte("DELETING..."))
	})

	http.ListenAndServe(":8000", r)
}
